<script>
/**
 * Opens a SweetAlert2 modal with an embedded video.
 * @param {string} url - The YouTube embed URL.
 * @param {string} title - The title of the video for accessibility.
 */
function openVideoModal(url, title) {
  const screenWidth = window.innerWidth || document.documentElement.clientWidth;
  const modalWidth = Math.min(0.9 * screenWidth, 900);

  Swal.fire({
    title: title,
    html: `
      <div class="swal-video-container">
        <iframe 
          src="${url}" 
          frameborder="0" 
          allow="autoplay; encrypted-media" 
          allowfullscreen 
          title="${title}">
        </iframe>
      </div>
    `,
    iconHtml: '<img style="width: 80px; height: 80px;" src="../img/IDS-logo.png" alt="IDS Logo">',
    width: modalWidth,
    showConfirmButton: true,
    confirmButtonText: 'Close',
    buttonsStyling: true,
    confirmButtonColor: '#3085d6',
    reverseButtons: true,
    footer: 'Questions? Contact IDS Support at <a href="mailto:support@thinkdataed.org">support@thinkdataed.org</a>',
    allowOutsideClick: true,
    // Swal handles focus trapping and "Escape to close" automatically
  });
}
</script>
